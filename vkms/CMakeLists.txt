cmake_minimum_required(VERSION 3.16)

project(driver VERSION 0.1.0 LANGUAGES C)
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Module info
add_definitions(-D__KERNEL__ -DMODULE)

# Find kernel build dir (allow override via -DKERNELHEADERS_DIR=...)
if (NOT DEFINED KERNELHEADERS_DIR)
    execute_process(COMMAND uname -r OUTPUT_VARIABLE KERNEL_RELEASE OUTPUT_STRIP_TRAILING_WHITESPACE)
    set(_possible_dirs
            /lib/modules/${KERNEL_RELEASE}/build
            /usr/src/kernels/${KERNEL_RELEASE}
            /usr/src/linux-headers-${KERNEL_RELEASE}
            /usr/src/linux-headers-${KERNEL_RELEASE}-generic
    )
    foreach (_d IN LISTS _possible_dirs)
        if (EXISTS "${_d}")
            set(KERNELHEADERS_DIR "${_d}")
            break()
        endif ()
    endforeach ()
endif ()

if (NOT DEFINED KERNELHEADERS_DIR OR NOT EXISTS "${KERNELHEADERS_DIR}")
    message(FATAL_ERROR "Could not find kernel build directory. Set -DKERNELHEADERS_DIR=/lib/modules/$(uname -r)/build")
endif ()
message(STATUS "Kernel build dir: ${KERNELHEADERS_DIR}")

include_directories("${KERNELHEADERS_DIR}/include")

# Collect sources under src/
file(GLOB_RECURSE SRC_C_FILES "${CMAKE_CURRENT_SOURCE_DIR}/src/*.c")
if (NOT SRC_C_FILES)
    message(FATAL_ERROR "No C sources found under ${CMAKE_CURRENT_SOURCE_DIR}/src")
endif ()

# Build object names (basenames -> .o)
set(OBJ_LIST "")
foreach (_src IN LISTS SRC_C_FILES)
    get_filename_component(_bn ${_src} NAME_WE)    # file name without extension
    list(APPEND OBJ_LIST "${_bn}.o")
endforeach ()

# Write a small Makefile into src/ for kbuild.
# It will list vkms-objs as the object list (kbuild is run from src/)
set(SRC_MAKEFILE "${CMAKE_CURRENT_SOURCE_DIR}/src/Makefile")
set(MAKEFILE_TEXT "# autogenerated by CMake for kbuild\n\n")
set(MAKEFILE_TEXT "${MAKEFILE_TEXT}vkms-objs := \\\n")

# Append all objects, avoid trailing backslash on last entry
list(LENGTH OBJ_LIST _count)
math(EXPR _last_idx "${_count} - 1")
foreach (idx RANGE 0 ${_last_idx})
    list(GET OBJ_LIST ${idx} _obj)
    if (idx EQUAL ${_last_idx})
        set(MAKEFILE_TEXT "${MAKEFILE_TEXT}    ${_obj}\n")
    else ()
        set(MAKEFILE_TEXT "${MAKEFILE_TEXT}    ${_obj} \\\n")
    endif ()
endforeach ()

set(MAKEFILE_TEXT "${MAKEFILE_TEXT}\nobj-$(CONFIG_DRM_VKMS) += vkms.o\nobj-$(CONFIG_DRM_VKMS_KUNIT_TEST) += tests/\n")

file(WRITE "${SRC_MAKEFILE}" "${MAKEFILE_TEXT}")
message(STATUS "Wrote kbuild Makefile to: ${SRC_MAKEFILE}")

# Where kbuild will put the module (kbuild is run in src/)
set(DRIVER_PATH "${CMAKE_CURRENT_SOURCE_DIR}/src/vkms.ko")

# Ensure we have a make program
if (NOT CMAKE_MAKE_PROGRAM)
    set(CMAKE_MAKE_PROGRAM make)
endif ()

# Provide full paths to source files for DEPENDS so CMake tracks them
set(SRC_FULL_DEPS "")
foreach (_s IN LISTS SRC_C_FILES)
    list(APPEND SRC_FULL_DEPS "${_s}")
endforeach ()

add_custom_command(
        OUTPUT "${DRIVER_PATH}"
        COMMAND "${CMAKE_MAKE_PROGRAM}" ARGS -C "${KERNELHEADERS_DIR}" "M=${CMAKE_CURRENT_SOURCE_DIR}/src" modules
        WORKING_DIRECTORY "${CMAKE_CURRENT_SOURCE_DIR}/src"
        DEPENDS ${SRC_FULL_DEPS} "${SRC_MAKEFILE}"
        VERBATIM
)

add_custom_target(driver ALL DEPENDS "${DRIVER_PATH}")
